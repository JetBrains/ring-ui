import Auth from 'ring-ui/components/auth/auth'; import hubConfig from 'ring-ui/site/hub-config'; import Http from 'ring-ui/components/http/http'; const auth = new Auth(hubConfig); const http = new Http(auth); async function initializeExample() { await auth.init(); const user = await http.get(`${hubConfig.serverUri}/api/rest/users/me?fields=name,login`); document.querySelector('#output').innerText = JSON.stringify(user); } initializeExample();
